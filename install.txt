Install pngquant on debian
==========================
# su
# apt-get install libpng-dev
# cd /tmp
# wget https://github.com/pornel/pngquant/archive/master.zip
# unzip master.zip
# cd pngquant-master
# make
# make install
# cd ..
# rm master.zip
# rm -rf pngquant-master
# ln -s /usr/local/bin/pngquant /bin/pngquant
